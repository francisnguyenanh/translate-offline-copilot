In app.py, add a new Flask route /proof-map (POST) with
@login_required decorator, placed after the existing /inject route.

This route:
  - Accepts the same file input as /inject (excel_file from session
    OR uploaded file, supporting .xlsx, .pptx, .docx)
  - Accepts pasted_json_data (JSON string) via form field
  - Accepts correction_color (hex string e.g. "#FF0000") via form field
  - Accepts dedup expansion (same expand_dedup_data call as /inject)
  - Writes the output file with BOTH the original text AND the
    AI-corrected text appended below it (in the correction color),
    only for keys where the corrected value differs from the original

Implement by calling three new helper functions (see below) instead of
the existing inject_* functions:

HELPER 1 — proof_map_xlsx(source_path, output_path, json_data, hex_color)

Uses openpyxl. For each key in json_data:
  1. Parse key as "SheetName!CellCoord" (same logic as extract for xlsx)
  2. Read the original cell value (orig_val = ws[coord].value)
  3. corrected_val = json_data[key]
  4. If str(orig_val).strip() == str(corrected_val).strip():
       → skip (no change)
     Else:
       → Set cell value to: orig_val + "\n" + corrected_val
       → Set cell font color for ONLY the correction portion is not
         possible at cell level with a simple newline — instead:
         use openpyxl's RichText approach via lxml to write an
         <is> element with two <r> runs:
           run 1: original text, original font color (preserve existing
                  font.color.rgb if available, else default)
           run 2: "\n" + corrected_val, font color = hex_color
         Write this as an inline string using cell's _value via
         the raw XML approach:
           from openpyxl.utils import get_column_letter
           from copy import deepcopy
           # Build <is><r><rPr><color rgb="FFXXXXXX"/></rPr><t>...</t></r>...</is>
           # and assign to cell via ws._cells[(row,col)]._value

HELPER 2 — proof_map_pptx(source_path, output_path, json_data, hex_color)

Uses python-pptx. For each key in json_data:
  1. Parse key using same logic as inject_text_to_pptx
  2. Find the target text_frame or table cell
  3. orig_text = current text of the shape/cell
  4. corrected_text = json_data[key]
  5. If orig_text.strip() == corrected_text.strip(): skip
     Else:
       a. Keep all existing paragraphs/runs intact (do not clear them)
       b. Add a NEW paragraph to the text_frame:
            new_para = text_frame.add_paragraph()
            new_run  = new_para.add_run()
            new_run.text = corrected_text
            from pptx.util import Pt
            from pptx.dml.color import RGBColor
            r, g, b = int(hex_color[1:3],16), int(hex_color[3:5],16), int(hex_color[5:7],16)
            new_run.font.color.rgb = RGBColor(r, g, b)
            # Copy font size from first run of first paragraph if available
            try:
              first_run = text_frame.paragraphs[0].runs[0]
              if first_run.font.size:
                new_run.font.size = first_run.font.size
            except Exception:
              pass

HELPER 3 — proof_map_docx(source_path, output_path, json_data, hex_color)

Uses python-docx. For each key in json_data:
  1. Parse key using same logic as inject_text_to_docx
     (ParagraphX, TableX!RyCz, Header_SectionX!ParagraphY, etc.)
  2. Find the target paragraph object
  3. orig_text = paragraph.text
  4. corrected_text = json_data[key]
  5. If orig_text.strip() == corrected_text.strip(): skip
     Else:
       a. Do NOT modify the original paragraph
       b. Insert a NEW paragraph immediately after it:
            from docx.oxml.ns import qn
            from docx.oxml import OxmlElement
            from docx.shared import RGBColor as DocxRGB
            # Use paragraph._element.addnext() to insert sibling
            new_para_elem = OxmlElement('w:p')
            new_run_elem  = OxmlElement('w:r')
            new_rpr       = OxmlElement('w:rPr')
            new_color     = OxmlElement('w:color')
            r,g,b = int(hex_color[1:3],16), int(hex_color[3:5],16), int(hex_color[5:7],16)
            new_color.set(qn('w:val'), f'{r:02X}{g:02X}{b:02X}')
            new_rpr.append(new_color)
            new_run_t = OxmlElement('w:t')
            new_run_t.text = corrected_text
            new_run_t.set('{http://www.w3.org/XML/1998/namespace}space', 'preserve')
            new_run_elem.append(new_rpr)
            new_run_elem.append(new_run_t)
            new_para_elem.append(new_run_elem)
            paragraph._element.addnext(new_para_elem)

@app.route('/proof-map', methods=['POST'])
@login_required
def proof_map():
    """
    Map kết quả kiểm tra ngữ pháp vào file gốc.
    Keys không thay đổi → giữ nguyên.
    Keys có sửa → thêm dòng mới với màu correction_color bên dưới.
    """
    # 1. Get source file (same as /inject: check session then upload)
    su_info = session.get('tab1_from_smart_update')
    if 'excel_file' in request.files and request.files['excel_file'].filename:
        excel_file = request.files['excel_file']
        use_session = False
        if not allowed_file(excel_file.filename):
            return jsonify({'error': 'Chỉ chấp nhận .xlsx, .pptx, .docx'}), 400
    elif su_info and os.path.exists(su_info.get('filepath', '')):
        use_session = True
        excel_file = None
    else:
        return jsonify({'error': 'Cần upload file gốc'}), 400

    # 2. Get correction color
    hex_color = request.form.get('correction_color', '#FF0000').strip()
    if not re.match(r'^#[0-9A-Fa-f]{6}$', hex_color):
        hex_color = '#FF0000'

    # 3. Get pasted JSON
    pasted_json_data = request.form.get('pasted_json_data', '').strip()
    if not pasted_json_data:
        return jsonify({'error': 'Cần paste JSON kết quả từ AI'}), 400
    try:
        json_data = json.loads(pasted_json_data)
    except json.JSONDecodeError as e:
        return jsonify({'error': f'JSON không hợp lệ: {e}'}), 400

    # 4. Expand dedup keys if present
    session_folder = get_session_folder()
    if any(k.startswith('dedup_') for k in json_data):
        json_data = expand_dedup_data(json_data, session_folder)

    # 5. Save source file to temp if uploaded
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    try:
        if not use_session:
            original_filename = excel_file.filename
            ext = original_filename.rsplit('.', 1)[1].lower()
            src_path = os.path.join(session_folder, f'proof_src_{timestamp}.{ext}')
            excel_file.save(src_path)
        else:
            original_filename = su_info['display_name']
            ext = original_filename.rsplit('.', 1)[1].lower()
            src_path = su_info['filepath']

        # 6. Output path
        base_name = os.path.splitext(original_filename)[0]
        out_display = f'{base_name}_proofread.{ext}'
        out_safe    = f'proof_out_{timestamp}.{ext}'
        out_path    = os.path.join(session_folder, out_safe)

        # 7. Dispatch to helper
        if ext == 'xlsx':
            proof_map_xlsx(src_path, out_path, json_data, hex_color)
            mimetype = ('application/vnd.openxmlformats-officedocument'
                        '.spreadsheetml.sheet')
        elif ext == 'pptx':
            proof_map_pptx(src_path, out_path, json_data, hex_color)
            mimetype = ('application/vnd.openxmlformats-officedocument'
                        '.presentationml.presentation')
        elif ext == 'docx':
            proof_map_docx(src_path, out_path, json_data, hex_color)
            mimetype = ('application/vnd.openxmlformats-officedocument'
                        '.wordprocessingml.document')
        else:
            return jsonify({'error': 'Định dạng không hỗ trợ'}), 400

        # 8. Store in session for download
        token = uuid.uuid4().hex[:12]
        session[f'proof_dl_{token}'] = {
            'path': out_path,
            'display_name': out_display,
            'mimetype': mimetype,
        }
        return jsonify({
            'success': True,
            'download_token': token,
            'display_name': out_display,
        })

    except Exception as e:
        app.logger.error(f'proof_map error: {e}', exc_info=True)
        return jsonify({'error': f'Lỗi xử lý: {str(e)}'}), 500


@app.route('/download-proof/<token>', methods=['GET'])
@login_required
def download_proof(token):
    """Serve file kết quả kiểm tra ngữ pháp."""
    key = f'proof_dl_{token}'
    if key not in session:
        return jsonify({'error': 'Token không hợp lệ hoặc đã hết hạn'}), 404
    info = session.pop(key)
    path = info['path']
    if not os.path.exists(path):
        return jsonify({'error': 'File không còn tồn tại'}), 404
    response = send_file(path, mimetype=info['mimetype'])
    response = set_download_headers(
        response, info['display_name'], 'proofread_output.' + path.rsplit('.',1)[-1])
    return response

// 7. Submit #proofMapForm → POST to /proof-map → download result

document.getElementById('proofMapForm').addEventListener('submit',
async function(e) {
  e.preventDefault();

  const color    = document.getElementById('proofCorrectionColor').value;
  const alertEl  = document.getElementById('proofMapAlert');
  const loading  = document.getElementById('proofMapLoading');
  const loadTxt  = document.getElementById('proofMapLoadingText');
  const btn      = document.getElementById('proofMapBtn');

  // Collect pasted JSON
  const accumulated = proofPastedJsonAccumulated;
  if (!Object.keys(accumulated).length) {
    alertEl.innerHTML =
      '<div class="alert alert-warning mt-2">Chưa có JSON nào được thêm.</div>';
    return;
  }

  alertEl.innerHTML = '';
  loading.style.display = 'block';
  loadTxt.textContent   = 'Đang ghi kết quả kiểm tra vào file...';
  btn.disabled = true;

  try {
    const fd = new FormData();

    // Re-attach source file from proofExcelFile (if still selected)
    // OR rely on session (same as Tab 1 inject behavior)
    const inp = document.getElementById('proofExcelFile');
    if (inp.files && inp.files.length) {
      // For multi-file, send first file as 'excel_file' for now
      // (single-file proof-map; batch can be extended later)
      fd.append('excel_file', inp.files[0]);
    }

    fd.append('pasted_json_data', JSON.stringify(accumulated));
    fd.append('correction_color', color);

    const resp = await fetch('/proof-map', { method: 'POST', body: fd });
    const data = await resp.json();

    if (!resp.ok || data.error) {
      alertEl.innerHTML =
        `<div class="alert alert-danger mt-2">${data.error}</div>`;
      return;
    }

    // Trigger download
    loadTxt.textContent = 'Đang chuẩn bị tải file...';
    const dlLink = document.createElement('a');
    dlLink.href = `/download-proof/${data.download_token}`;
    dlLink.download = data.display_name;
    document.body.appendChild(dlLink);
    dlLink.click();
    document.body.removeChild(dlLink);

    alertEl.innerHTML =
      `<div class="alert alert-success mt-2">
         <i class="fas fa-check-circle"></i>
         Đã tạo file <strong>${data.display_name}</strong>.
         Đang tải về...
       </div>`;

    // Mark step 3 done
    document.getElementById('proofBadge3').innerHTML =
      '<i class="fas fa-check" style="font-size:0.8rem"></i>';
    document.getElementById('proofBadge3').style.background =
      'linear-gradient(135deg,#28a745,#20c997)';

    // Reset accumulated JSON for next run
    proofPastedJsonAccumulated = {};
    renderProofPastedJsonList();
    document.getElementById('proofMapBtn').disabled = true;

  } catch(err) {
    alertEl.innerHTML =
      `<div class="alert alert-danger mt-2">Lỗi kết nối: ${err.message}</div>`;
  } finally {
    loading.style.display = 'none';
    btn.disabled = false;
  }
});
